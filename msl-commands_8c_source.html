137/* Change registers are memory-mapped, 16-bit big-endian, at addresses
138 * 240 + N*2 (N = 0..7). The register-op aux char (in L1) carries the 4-bit
139 * register code 1000..1111; the register index N = code & 7.
140 * NOTE: the code is assumed to sit in the LOW nibble of the aux char —
141 * verify against the funktionalcpu oracle. V1 holds the I1 memory address. */
142static uint16_t reg_addr_of(struct ge* ge) { return (uint16_t)(240 + (ge->rL1 & 0x07) * 2); }
143static uint16_t cr_rd16(struct ge* ge, uint16_t a) {
144 return (uint16_t)((ge->mem[a] << 8) | ge->mem[(uint16_t)(a + 1)]);
145}
146static void cr_wr16(struct ge* ge, uint16_t a, uint16_t v) {
147 ge->mem[a] = (uint8_t)(v >> 8);
148 ge->mem[(uint16_t)(a + 1)] = (uint8_t)(v & 0xff);
149}

151/* Address modification — bit 15 = absolute/modified flag (CPU[4] §2.5, FO.19-20;
152 * flow chart dwg 14023130, transcribed in reference_operand_fetch_flowchart).
153 *
154 * Every operand address field carries the modify flag in bit 15:
155 * bit 15 = 0 ABSOLUTE: bits 0-14 are the address directly (0..0x7FFF); the
156 * change registers are NOT consulted.
157 * bit 15 = 1 MODIFIED: bits 12-14 select one of the 8 change registers and
158 * EA = change_register[N] + (bits 0-11 displacement), 16-bit wrap.
159 *
160 * The GE-130 resolves this *during operand fetch*: an absolute field is left in
161 * V verbatim (bit 15 = 0, so V < 0x8000); a modified field is reduced to its
162 * 12-bit displacement (top quartet zeroed) in V and the change register is then
163 * added by the indexing micro-cycle ED|EC -> EF|EE (EXEC_INDEX below), which
164 * leaves the resolved EA in V2 and (for the first operand, SA00) copies it to V1.
165 * So after fetch the operand registers hold the RESOLVED effective address and
166 * execution uses them directly — eff_v1_l2 / CI00s / EXEC_SS just mask to the
167 * 15-bit address space. The change registers live at mem[240+2N] and default to
168 * base[N] = N<<12 at reset (ge_clear); programs reload a base via LR/LA/AMR. */
169
170/* Read change register N (16-bit big-endian at mem[240+2N]). */
171static uint16_t cr_base(struct ge* ge, int n) {
172 uint16_t a = (uint16_t)(240 + (n & 7) * 2);
173 return (uint16_t)((ge->mem[a] << 8) | ge->mem[(uint16_t)(a + 1)]);
174}

186/* Modified-address indexing (flow chart ED|EC, the low-byte add, fused with the
187 * EF|EE high-byte/carry add into one hybrid step). The displacement sits in
188 * V2 bits 0-11 (top quartet was zeroed during the high-byte read for a modified
189 * address); the modifier N is L2 bits 4-6 (= address bits 12-14, since L2 holds
190 * the current operand's addr-hi byte). EA = change_register[N] + displacement,
191 * 16-bit wrap (carry beyond word 2 lost). The resolved EA is written to V2 and,
192 * for the first operand (SA00), copied to V1. */
193static void EXEC_INDEX(struct ge* ge) {
194 uint8_t n = (ge->rL2 >> 4) & 7;
195 uint16_t disp = ge->rV2 & 0x0FFFu;
196 uint16_t ea = (uint16_t)(cr_base(ge, n) + disp);
197 ge->rV2 = ea;
198 if (ge->SA00)
199 ge->rV1 = ea;
200}

220/* SS (Storage-to-Storage) data-op execution commands (Wave 5 / Mechanism B).
221 *
222 * EXEC_SS fires from state_E7 at TI06, after CI02 at TI05 has loaded V2
223 * (source address) and V1 (destination address) was already set by the
224 * preceding E5 pass. SS_TO_ALPHA then overrides the future state to
225 * 0xe2 (alpha), breaking out of the operand-fetch micro-loop.
226 *
227 * Operand layout at TI06 of state_E7:
228 * V1 = destination address
229 * V2 = source address
230 * L1 = length byte
231 *
232 * Single-length ops (MVC/NC/OC/XC/CMC/TL/UPK/PK/EDT):
233 * len = (rL1 & 0xff) + 1
234 *
235 * Two-length decimal ops (AP/SP/MP/DP/MVP/CMP/PKS/UPKS):
236 * alen = ((rL1 >> 4) & 0xf) + 1 (high nibble + 1)
237 * blen = (rL1 & 0xf) + 1 (low nibble + 1)
238 */
239static void EXEC_SS(struct ge *ge)
240{
241 uint8_t len = (ge->rL1 & 0xff) + 1;
242 uint8_t alen = ((ge->rL1 >> 4) & 0xf) + 1;
243 uint8_t blen = (ge->rL1 & 0x0f) + 1;
244 /* SS effective addresses: V1 (dest) and V2 (src) already hold the resolved
245 * effective address after operand fetch (absolute field verbatim, or
246 * change_register+displacement via the indexing micro-cycle). Mask to the
247 * 15-bit address space. */
248 uint16_t dst = ge->rV1 & 0x7FFFu;
249 uint16_t src = ge->rV2 & 0x7FFFu;
250
251 switch (ge->rFO) {
252 case MVC_OPCODE:
253 alu_mvc(ge, dst, src, len);
254 break;
255 case NC_OPCODE:
256 alu_nc(ge, dst, src, len);
257 break;
258 case OC_OPCODE:
259 alu_oc(ge, dst, src, len);
260 break;
261 case XC_OPCODE:
262 alu_xc(ge, dst, src, len);
263 break;
264 case CMC_OPCODE:
265 alu_cmc(ge, dst, src, len);
266 break;
267 case TL_OPCODE:
268 alu_tl(ge, dst, len, src);
269 break;
270 case UPK_OPCODE:
271 alu_upk(ge, dst, len-1, src, len-1);
272 break;
273 case PK_OPCODE:
274 alu_pk(ge, dst, len-1, src, len-1);
275 break;
276 case EDT_OPCODE:
277 alu_edt(ge, dst, len, src);
278 break;
279 case MVP_OPCODE:
280 alu_mvp(ge, dst, alen-1, src, blen-1);
281 break;
282 case CMP_OPCODE:
283 alu_cmp(ge, dst, alen-1, src, blen-1);
284 break;
285 case AP_OPCODE:
286 alu_ap(ge, dst, alen-1, src, blen-1);
287 break;
288 case SP_OPCODE:
289 alu_sp(ge, dst, alen-1, src, blen-1);
290 break;
291 case MP_OPCODE:
292 alu_mp(ge, dst, alen-1, src, blen-1);
293 break;
294 case DP_OPCODE:
295 alu_dp(ge, dst, alen-1, src, blen-1);
296 break;
297 case PKS_OPCODE:
298 alu_pks(ge, dst, alen-1, src, blen-1);
299 break;
300 case UPKS_OPCODE:
301 alu_upks(ge, dst, alen-1, src, blen-1);
302 break;
303 case AB_OPCODE:
304 alu_ab(ge, dst, alen, src, blen);
305 break;
306 case SB_OPCODE:
307 alu_sb(ge, dst, alen, src, blen);
308 break;
309 case AD_OPCODE:
310 alu_ad(ge, dst, alen, src, blen);
311 break;
312 case SD_OPCODE:
313 alu_sd(ge, dst, alen, src, blen);
314 break;
315 case MVQ_OPCODE:
316 alu_mvq(ge, dst, src, len);
317 break;
318 case CMQ_OPCODE:
319 alu_cmq(ge, dst, src, len);
320 break;
321 default:
322 ge_log(LOG_ERR, "EXEC_SS: unknown SS opcode 0x%02x\n", ge->rFO);
323 break;
324 }

334/* EPER "examine abnormal conditions": load the channel-1 peripheral status
335 * into RO so the qualitative-result decode (DU95 = !RO1 && !RO2 && RO6 = "no
336 * error") reflects the real status. Our integrated-reader feed is always
337 * clean (no parity/format error; CE02 even ignores the parity check), so the
338 * status is no-error: RO6 set, error bits (RO1/RO2) clear. A real error would
339 * be reported here once error injection is modelled. */
340static void CE_chan1_status(struct ge *ge)
341{
342 /* Default: RO6=1 (operation OK), RO1=RO2=0 (no error) -> DU95=1 ("no
343 * error"). When inject_chan1_status is non-zero a test/harness is injecting
344 * an abnormal condition: report that status byte instead (e.g. 0x42 sets
345 * RO1, so DU95 reads "error" and the EPER examine branch fires). */
346 ge->rRO = ge->inject_chan1_status ? ge->inject_chan1_status : 0x40;
347}
