51/* Seed the eight change / segment-base registers to their identity defaults
52 * N<<12: change register N is the 16-bit big-endian word at mem[240+2N], and
53 * an instruction address with modifier N (address bits 12-14) resolves to
54 * displacement + base[N]. With these defaults a bare 12-bit displacement
55 * carrying modifier N addresses segment N (0x1000*N ..), so a program's paged
56 * addresses (e.g. JU 0x172a) resolve to their full load addresses; programs
57 * may reload a base via LR/LA for paged access.
58 *
59 * Called by ge_clear (reset) and re-applied after a direct binary image load,
60 * because a contiguous image spanning the 0x00F0-0x00FF window would otherwise
61 * overwrite the bases (with its own bytes, or zeros in reconstructed gaps). */
62void ge_seed_segment_bases(struct ge *ge)
63{
64 for (int n = 0; n < 8; n++) {
65 uint16_t v = (uint16_t)(n << 12);
66 ge->mem[240 + 2 * n] = (uint8_t)(v >> 8);
67 ge->mem[240 + 2 * n + 1] = (uint8_t)(v & 0xff);
68 }
69}

84/* odd-parity bit for a byte: 1 if the byte has an even number of set bits
85 * (so data+parity is odd). Mirrors odd_parity() in pulse.c. */
86static inline uint8_t ge_odd_parity(uint8_t data)
87{
88 return __builtin_parity(data) ? 0 : 1;
89}

91/* Load a flat image into memory at `origin` (the unified-format payload).
92 * Unlike ge_load_program this is origin-aware and not size-capped; it also
93 * primes the parity store and marks the cells written, so reads of the loaded
94 * code parity-check cleanly. Returns 0 on success, -1 if it would exceed the
95 * installed memory. */
96int ge_load_image(struct ge *ge, const uint8_t *image, size_t size,
97 uint16_t origin)
98{
99 uint32_t max = ge->mem_size ? ge->mem_size : MEM_SIZE;
100
101 if (image == NULL && size != 0)
102 return -1;
103 if ((uint32_t)origin + (uint32_t)size > max)
104 return -1;
105
106 for (size_t i = 0; i < size; i++) {
107 uint16_t a = (uint16_t)(origin + i);
108 ge->mem[a] = image[i];
109 ge->mem_parity[a] = ge_odd_parity(image[i]);
110 ge->mem_written[a] = 1;
111 }
112 return 0;
113}

115/* Enter execution at `entry` without the peripheral LOAD bootstrap: seed the
116 * program counter and drop the sequencer straight into the alpha (fetch) phase.
117 * Use after ge_clear + ge_load_image for the direct binary-load path; the
118 * --deck card-reader path uses the natural state 00 -> 80 -> alpha bootstrap
119 * instead (which leaves entry at 0). */
120void ge_enter(struct ge *ge, uint16_t entry)
121{
122 ge->rPO = entry;
123 ge->rSO = 0xe2; /* alpha phase: fetch the instruction at PO */
124 ge->rSA = 0xe2;
125}

258int ge_run_pulse(struct ge *ge)
259{
260 int r;
261 struct msl_timing_state *state;
262
263 if (ge_clock_is_first(ge)) {
264 r = ge_peri_on_clock(ge);
265 if (r != 0)
266 return r;
267
268 /* poll the connectors and try to set up the cpu state.
269 * should this be here? */
270 connectors_first_clock(ge);
271 }
272
273 /* Execute common pulse machine logic */
274 pulse(ge);
275
276 /* Execute peripherals pulse callbacks */
277 r = ge_peri_on_pulses(ge);
278 if (r != 0)
279 return r;
280
281 /* Execute the commands from the timing charts */
282 state = msl_get_state(ge->rSA);
283
284 /* The state to execute gets loaded in SA at TO10 */
285 if (ge->current_clock == TO10)
286 ge_print_well_known_states(ge->rSA);
287
288 if (!state) {
289 ge_log(LOG_ERR, "no timing charts found for state %02X\n", ge->rSA);
290 return 1;
291 }
292
293 msl_run_state(ge, state);
294
295 if (ge_clock_is_last(ge)) {
296 fsn_last_clock(ge);
297 ge_print_registers_nonverbose(ge);
298 }
299
300 ge_clock_increment(ge);
301 return 0;
302}

329void fsn_last_clock(struct ge *ge)
330{
331 /* At the end of a CPU cycle the future-status network is stored in SO
332 * (cpu fo. 127), advancing the program sequencer one state.
333 *
334 * In maintenance forcing (rotary off NORM) the program sequencer is frozen:
335 * the manual (CPU[4] §4 "Maintenance Panel", dwg 30004122 fo. 35-37) says a
336 * forcing cycle writes the *register under exam* (displayed through BO), it
337 * does not step the program. So a forcing cycle must NOT advance SO — that
338 * is what lets the operator key an instruction across phases (force SO=E2,
339 * step to E0, force FO, step to the 0x64 beta, force L1, step to execute).
340 * The one exception is rotary position 13 (RS_SO), which forces SO/SI
341 * itself — that is how the operator sets the sequencer state.
342 *
343 * RICI ("disable next status") suppresses the advance in normal operation,
344 * letting a status be re-executed. */
345 uint8_t sel_norm = ge->register_selector == RS_NORM;
346 uint8_t sel_so = ge->register_selector == RS_SO;
347 uint8_t advance_so = sel_norm ? !ge->console_switches.RICI : sel_so;
348 if (ge->RIA0 && advance_so) {
349 ge_log(LOG_FUTURE, "last clock cpu, %02x in SO\n", ge->future_state);
350 ge->rSO = ge->future_state;
351 } else {
352 ge_log(LOG_FUTURE, "last clock cpu, SO held at %02x (RIA0 %d advance %d)\n",
353 ge->rSO, ge->RIA0, advance_so);
354 }
355
356 /* after the end of a cpu work cycle, (ALTO / ALS71=1) is set if
357 * the PAPA switch is inserted, or if the rotary switch is neither
358 * in the normal position, nor in position 8 for recording in
359 * memory ALSOA=0) (cpu fo. 98)
360 */
361 uint8_t is_papa = ge->console_switches.PAPA;
362 uint8_t is_norm = ge->register_selector == RS_NORM;
363 uint8_t is_scr = ge->register_selector == RS_V1_SCR;
364 ge_log(LOG_FUTURE, " papa: %d, norm: %d, scr: %d ==> %d\n", is_papa, is_norm, is_scr, ge->RIA0 && (is_papa || !(is_norm || is_scr)));
365
366 if (ge->RIA0 && (is_papa || !(is_norm || is_scr)))
367 ge->ALTO = 1;
368
369 /* after the execution of a channel 2 cycle, load the first
370 * 4 bits of the future status network in SI. (cpu fo. 127) */
371 if (ge->RIA2) {
372 ge_log(LOG_FUTURE, "last clock ch2, %02x in SI\n", ge->future_state);
373 ge->rSI = ge->future_state;
374 }
