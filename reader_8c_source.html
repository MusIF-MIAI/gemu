36void reader_setup_to_send(struct ge *ge, uint8_t data, uint8_t end)
37{
38 ge->integrated_reader.lu08 = 1;
39 ge->integrated_reader.data = data;
40 ge->integrated_reader.fini = end;
41
42 /* When end=1, set the end-of-transfer flip-flops.
43 *
44 * RIG1 is the "reader end" flip-flop; the real hardware sets it via
45 * the RF101 signal chain (FINI1 && PC121). Here we set it directly
46 * whenever the peripheral signals end-of-card (fini=1), which is
47 * equivalent because we only call reader_setup_to_send from the
48 * integrated-reader path (PC121=1).
49 *
50 * PEC1 is the "peripheral end complete" flip-flop; the real hardware
51 * sets it via PIM11 (end-of-transfer strobe) once RF101 is asserted.
52 * PIM11 depends on TO50 and several channel-status signals that are
53 * not yet fully modelled; the direct set here is the working
54 * approximation until PIM11/RS011 are complete.
55 */
56 if (end) {
57 ge->RIG1 = 1;
58 ge->PEC1 = 1;
59 }
60
61 if (RB111(ge)) {
62 ge_log(LOG_READER, "XXX\n");
63 }
64}

66void reader_clear_sending(struct ge *ge)
67{
68 ge->integrated_reader.lu08 = 0;
69 ge->integrated_reader.data = 0;
70 /* Clear the end-of-card strobe so FINI1 / RF101 deassert once the
71 * machine has consumed the end byte. This is essential for multi-card
72 * reads: if fini stays 1, the NEXT card's first byte would immediately
73 * signal end-of-card again before any data is read. */
74 ge->integrated_reader.fini = 0;
75}
