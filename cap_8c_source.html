1/*
2 * cap.c — parser for Raspberry-Pi-Pico punch-card reader .cap files.
3 *
4 * File format:
5 * - Free-text preamble lines, bare column counters, "FEED ON/OFF" etc.
6 * - Card delimiter: a line of the exact form "Card n. N" (N = 1, 2, 3, …)
7 * - Following the delimiter: whitespace/newline-separated 4-hex-digit tokens,
8 * one per card column (80 per card), each representing a 12-bit hole pattern.
9 * - Non-hex lines between card data are ignored.
10 *
11 * This parser is fully reentrant; no global state.
12 */
13

42/* Grow the card array in the deck by one slot. Returns 0 on success. */
43static int deck_add_card(struct cap_deck *d)
44{
45 if (d->ncards >= d->cap) {
46 int newcap = d->cap ? d->cap * 2 : 16;
47 struct cap_card *tmp = realloc(d->cards,
48 (size_t)newcap * sizeof(struct cap_card));
49 if (!tmp)
50 return -1;
51 d->cards = tmp;
52 d->cap = newcap;
53 }
54 d->cards[d->ncards].cols = NULL;
55 d->cards[d->ncards].ncols = 0;
56 d->cards[d->ncards].cap = 0;
57 d->ncards++;
58 return 0;
59}

61/* Append one column value to the current (last) card. Returns 0 on success. */
62static int card_add_col(struct cap_card *c, uint16_t val)
63{
64 if (c->ncols >= c->cap) {
65 int newcap = c->cap ? c->cap * 2 : 80;
66 uint16_t *tmp = realloc(c->cols, (size_t)newcap * sizeof(uint16_t));
67 if (!tmp)
68 return -1;
69 c->cols = tmp;
70 c->cap = newcap;
71 }
72 c->cols[c->ncols++] = val & 0x1FFFu; /* keep 13 bits (12-bit hole + spare) */
73 return 0;
74}

76/*
77 * Return non-zero if `s` is a valid 4-hex-digit token (exactly 4 chars,
78 * all hexadecimal digits).

80static int is_hex4(const char *s)
81{
82 if (!s || strlen(s) != 4)
83 return 0;
84 return isxdigit((unsigned char)s[0]) &&
85 isxdigit((unsigned char)s[1]) &&
86 isxdigit((unsigned char)s[2]) &&
87 isxdigit((unsigned char)s[3]);
88}

90/*
91 * Return non-zero if line begins with "Card n. " (case-sensitive).
92 * If it does, *card_num receives the card number (>= 1).
93 */
94static int parse_card_header(const char *line, int *card_num)
95{
96 const char prefix[] = "Card n. ";
97 const size_t plen = sizeof(prefix) - 1;
98 if (strncmp(line, prefix, plen) != 0)
99 return 0;
100 const char *p = line + plen;
101 /* Require at least one digit immediately after the prefix */
102 if (!isdigit((unsigned char)*p))
103 return 0;
104 char *end;
105 long n = strtol(p, &end, 10);
106 if (n < 1)
107 return 0;
108 /* Allow trailing whitespace/newline only */
109 while (*end && (isspace((unsigned char)*end) || *end == '\r'))
110 end++;
111 if (*end != '\0')
112 return 0;
113 *card_num = (int)n;
114 return 1;
115}

121struct cap_deck *cap_load(const char *path)
122{
123 if (!path)
124 return NULL;
125
126 FILE *fp = fopen(path, "r");
127 if (!fp)
128 return NULL;
129
130 struct cap_deck *d = calloc(1, sizeof(struct cap_deck));
131 if (!d) {
132 fclose(fp);
133 return NULL;
134 }
135
136 int in_card = 0; /* whether we are inside a card's data section */
137
138 char line[4096];
139 while (fgets(line, sizeof(line), fp)) {
140 /* Strip trailing newline/CR */
141 size_t len = strlen(line);
142 while (len > 0 && (line[len-1] == '\n' || line[len-1] == '\r'))
143 line[--len] = '\0';
144
145 /* Check for card header */
146 int card_num = 0;
147 if (parse_card_header(line, &card_num)) {
148 if (deck_add_card(d) != 0)
149 goto oom;
150 in_card = 1;
151 continue;
152 }
153
154 /* If we are not yet inside any card, skip non-card lines */
155 if (!in_card)
156 continue;
157
158 /*
159 * We are inside a card data section. Tokenize the line and
160 * collect 4-hex-digit tokens; ignore anything else (counters,
161 * "FEED ON/OFF", blank lines, etc.).
162 */
163 char *buf = line;
164 char *tok;
165 /* Use a copy because strtok modifies its argument */
166 char linecopy[4096];
167 strncpy(linecopy, line, sizeof(linecopy) - 1);
168 linecopy[sizeof(linecopy) - 1] = '\0';
169
170 tok = strtok(linecopy, " \t\r\n");
171 while (tok) {
172 if (is_hex4(tok)) {
173 uint16_t val = (uint16_t)strtoul(tok, NULL, 16);
174 struct cap_card *cur = &d->cards[d->ncards - 1];
175 if (card_add_col(cur, val) != 0)
176 goto oom;
177 }
178 tok = strtok(NULL, " \t\r\n");
179 }
180 (void)buf; /* suppress unused-variable warning */
181 }
182
183 fclose(fp);
184 return d;
185
186oom:
187 fclose(fp);
188 cap_free(d);
189 return NULL;
190}

227static int scat_decode_card(const uint16_t *cols, int ncols, int mode,
228 uint8_t out[80])

238/* Detect the deck's dominant 8-byte data-card prefix (cols 0-7). Returns the
239 * match count and copies it into out[8]; 0 if no eligible cards. */
240static int scat_detect_prefix(struct cap_deck *d, int mode, uint8_t out[8])
241{
242 int nc = cap_num_cards(d);
243 struct { uint8_t p[8]; int cnt; } tab[1024];
244 int ntab = 0, best = -1;
245
246 for (int i = 0; i < nc; i++) {
247 int ncols = cap_card_ncols(d, i);
248 const uint16_t *cols = cap_card_columns(d, i);
249 if (ncols < 11 || !cols)
250 continue;
251
252 uint8_t b[80];
253 scat_decode_card(cols, ncols, mode, b);
254
255 int j;
256 for (j = 0; j < ntab; j++) {
257 if (memcmp(tab[j].p, b, 8) == 0) {
258 tab[j].cnt++;
259 break;
260 }
261 }
262 if (j == ntab && ntab < 1024) {
263 memcpy(tab[ntab].p, b, 8);
264 tab[ntab].cnt = 1;
265 ntab++;
266 }
267 }
268
269 for (int j = 0; j < ntab; j++) {
270 if (best < 0 || tab[j].cnt > tab[best].cnt)
271 best = j;
272 }
273 if (best < 0)
274 return 0;
275
276 memcpy(out, tab[best].p, 8);
277 return tab[best].cnt;
278}

280int cap_load_scattered(const char *path, int mode, unsigned char *image,
281 unsigned *lo, unsigned *hi)
282{
283 struct cap_deck *d = cap_load(path);
284 if (!d)
285 return -1;
286
287 uint8_t want[8];
288 int have_prefix = (scat_detect_prefix(d, mode, want) >= 4);
289 int loose = !have_prefix; /* no dominant prefix: accept any fitting record */
290
291 int nc = cap_num_cards(d);
292 int loaded = 0;
293 long min_a = -1, max_a = -1;
294
295 for (int i = 0; i < nc; i++) {
296 int ncols = cap_card_ncols(d, i);
297 const uint16_t *cols = cap_card_columns(d, i);
298 if (ncols < 11 || !cols)
299 continue;
300
301 uint8_t b[80];
302 int n = scat_decode_card(cols, ncols, mode, b);
303
304 int match = (have_prefix && n >= 8 && memcmp(b, want, 8) == 0);
305 int ll = b[8];
306 long addr = ((long)b[9] << 8) | b[10];
307 int paylen = ll + 1;
308 int fits = (11 + ll < n) && (addr + ll <= 0xFFFF);
309
310 if (!(loose ? fits : (match && fits)))
311 continue;
312
313 for (int k = 0; k < paylen; k++)
314 image[addr + k] = b[11 + k];
315
316 if (min_a < 0 || addr < min_a)
317 min_a = addr;
318 if (addr + ll > max_a)
319 max_a = addr + ll;
320 loaded++;
321 }
322
323 cap_free(d);
324
325 if (loaded == 0 || min_a < 0)
326 return -1;
327
328 if (lo)
329 *lo = (unsigned)min_a;
330 if (hi)
331 *hi = (unsigned)max_a;
332 return loaded;
333}
