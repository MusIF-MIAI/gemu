15static inline uint8_t odd_parity(uint8_t data)
16{
17 /* __builtin_popcount gives the number of set bits; if it's already
18 * odd we need parity=0; if even we need parity=1. */
19 return (__builtin_popcount(data) & 1) ? 0 : 1;
20}

97static void on_TO50(struct ge *ge) {
98 /* not sure about the timing of memory ops
99 * read was previously done in TO65 with write, but
100 * it didn't work to implement the state CC for PERI.
101 * reading here seems to work in all known cases */
102 if (ge->memory_command == MC_READ) {
103 uint32_t size = ge->mem_size ? ge->mem_size : MEM_SIZE;
104
105 if (ge->rVO >= size) {
106 /* Address outside installed memory: raise INV ADD fault */
107 ge->inv_add = 1;
108 ge_log(LOG_STATES, "memory read: INV ADD rVO=%x >= size=%x\n",
109 ge->rVO, size);
110 } else {
111 ge->rRO = ge->mem[ge->rVO];
112 ge_log(LOG_STATES, "memory read: RO = mem[VO] = mem[%x] = %x\n",
113 ge->rVO, ge->rRO);
114
115 /* Parity check: only for locations that have been written.
116 * Cleared/uninitialised locations are not checked to avoid
117 * false MEM CHECK on startup. */
118 if (ge->mem_written[ge->rVO]) {
119 if (odd_parity(ge->mem[ge->rVO]) != ge->mem_parity[ge->rVO]) {
120 ge->mem_check = 1;
121 ge_log(LOG_STATES,
122 "memory read: MEM CHECK parity error at %x\n",
123 ge->rVO);
124 }
125 }
126 }
127
128 ge->memory_command = MC_NONE;
129 }
130}
