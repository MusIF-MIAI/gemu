1/*
2 * printer.h - Integrated printer / console-typewriter peripheral (channel 2).
3 *
4 * Pragmatic model: gemu does not drive the channel-2 (rSI) per-character
5 * transfer states at signal level, so a print/typewriter PER leaves the CPU
6 * suspended in the org-phase external request-wait (state b8, rSA idle). This
7 * peripheral detects that stall and completes the operation the way the real
8 * channel-2 unit would — by asserting the unit-ready condition (PUC2, which
9 * makes DU97 true) and the CPU-active request (RC00) so the machine's OWN
10 * state_b8 microcode (CU01/CU13/CU14/CU06) routes B8 -> alpha with the CPU
11 * context intact. No state is forced from outside.
12 *
13 * Capture is best-effort: at completion the print order block referenced by the
14 * channel-2 operand addresser (rV2) is read into the paper-feed buffer, each
15 * byte rendered through the GE 100-series internal graphic set (gecode.c) — the
16 * machine's own character code, not ASCII. The per-character channel-2 transfer
17 * itself is not modelled, so what rV2 points at may be the order block rather
18 * than the text buffer (e.g. the funktionalcpu PER -> non-graphic control bytes,
19 * rendered '.'); a faithful line needs the channel-2 transfer states wired.
20 *
21 * Two-way: kbd[] is the operator-keyboard input queue (printer_feed_key); the
22 * wasm/interactive front-end pushes typed characters here.
23 */

25#define PRINTER_H

35/* Begin a channel-2 OUTPUT transfer: print `length` characters starting at
36 * memory address `buffer`. The printer then requests a channel-2 cycle per
