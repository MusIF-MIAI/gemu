26/*
27 * Launch the ncurses console client (console/curses/console.py) as a child
28 * process for --tui. The client connects to the /tmp/gemu.console socket that
29 * --console registers and draws the operator/diagnostic panel; the emulator
30 * keeps running in this (parent) process. Returns the child pid, or -1 if the
31 * client could not be found / launched.
32 *
33 * The script is looked for next to the ge executable first (so it works from
34 * any cwd), then relative to the current directory.

36/* Interactive console switches driven by signals: SIGUSR1 toggles SWITCH 1
37 * (JS1), SIGUSR2 toggles SWITCH 2 (JS2). The handler only sets a flag; the
38 * run loop applies it between cycles (so we never touch ge state from a
39 * handler). Lets a human (or an automated harness) flip the diagnostic
40 * switches mid-run: e.g. start the funktionalcpu test with SWITCH 2 on, then
41 * `kill -USR2 <pid>` to release it and watch where the deck goes. */
42static volatile sig_atomic_t g_toggle_js1 = 0;
43static volatile sig_atomic_t g_toggle_js2 = 0;
44static void on_sigusr1(int sig) { (void)sig; g_toggle_js1 = 1; }
45static void on_sigusr2(int sig) { (void)sig; g_toggle_js2 = 1; }

47static pid_t spawn_tui(const char *argv0)
48{
49 char path[4096];
50 const char *slash = strrchr(argv0, '/');
51
52 if (slash) {
53 int dlen = (int)(slash - argv0);
54 snprintf(path, sizeof(path), "%.*s/console/curses/console.py", dlen, argv0);
55 } else {
56 snprintf(path, sizeof(path), "console/curses/console.py");
57 }
58 if (access(path, R_OK) != 0)
59 snprintf(path, sizeof(path), "console/curses/console.py");
60 if (access(path, R_OK) != 0) {
61 fprintf(stderr, "error: --tui: cannot find console/curses/console.py\n");
62 return -1;
63 }
64
65 pid_t pid = fork();
66 if (pid < 0) {
67 perror("fork");
68 return -1;
69 }
70 if (pid == 0) {
71 execlp("python3", "python3", path, (char *)NULL);
72 perror("error: --tui: cannot exec python3");
73 _exit(127);
74 }
