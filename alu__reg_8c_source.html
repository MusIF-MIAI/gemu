64/* =========================================================================
65 * CMR — Compare Memory to Register (0xBD)
66 * §5.6.4.4: purely qualitative result, no operand modification.
67 *
68 * CC encoding (FA04=bit4, FA05=bit5 of ffFI; cc=(FA04<<1)|FA05):
69 * FA04 FA05 cc meaning
70 * 0 0 0 "Not possible" — never set
71 * 0 1 1 Register < Memory → ALU_CC_LOW (1)
72 * 1 0 2 Register == Memory → ALU_CC_EQUAL (2)
73 * 1 1 3 Register > Memory → ALU_CC_HIGH (3)
74 *
75 * Confidence: HIGH (direct table from §5.6.4.4 and confirmed in §5.10.2)
76 * =========================================================================*/
77void alu_cmr(struct ge *ge, uint16_t reg_val, uint16_t value)
78{
79 uint8_t cc;
80
81 if (reg_val < value)
82 cc = ALU_CC_LOW; /* 1 */
83 else if (reg_val == value)
84 cc = ALU_CC_EQUAL; /* 2 — "Reg equal to memory" maps to FA04=1,FA05=0 */
85 else
86 cc = ALU_CC_HIGH; /* 3 — "Reg greater than memory" */
87
88 alu_set_cc(ge, cc);
89}

91/* =========================================================================
92 * AMR — Add Memory to Register (0xBE)
93 * §5.6.4.2: register = register + memory_value; result left in register.
94 * Performed as 16-bit binary arithmetic; carry beyond 16 bits is lost
95 * ("overflow").
96 *
97 * CC encoding (§5.6.4.2) — ADD-magnitude table:
98 * FA04 FA05 cc meaning
99 * 0 0 0 result == 0, no overflow (raw 0)
100 * 0 1 1 result != 0, no overflow (raw 1)
101 * 1 0 2 overflow, partial==0 (raw 2)
102 * 1 1 3 overflow, partial!=0 (raw 3)
103 *
104 * Carry flip-flops:
105 * URPE = carry out of bit 15 (the "overflow" carry, bit 16 of 32-bit sum)
106 * URPU = carry out of bit 14 (carry into bit 15)
107 *
108 * Confidence: HIGH (direct table §5.6.4.2)
109 * =========================================================================*/
110void alu_amr(struct ge *ge, uint16_t *reg, uint16_t value)
111{
112 uint32_t full = (uint32_t)(*reg) + (uint32_t)value;
113 uint16_t result = (uint16_t)(full & 0xFFFF);
114
115 /* Carry out of bit 15 → overflow */
116 uint8_t carry_out = (full >> 16) & 1;
117
118 /*
119 * Carry out of bit 14 = carry into bit 15.
120 * This is the carry from the low 15 bits into bit 15:
121 * bit-15 of the sum is produced by (a[14] + b[14] + carry_into_14).
122 * We can detect it as: if the top bit of result differs from
123 * what a plain 15-bit addition would give.
124 * Simpler: carry_into_15 = carry out of bit 14 = ((a & 0x7FFF) + (b & 0x7FFF)) >> 15
125 */
126 uint8_t carry_into_15 = (uint8_t)(
127 (((uint32_t)(*reg & 0x7FFF)) + ((uint32_t)(value & 0x7FFF))) >> 15
128 );
129
130 ge->URPE = carry_out;
131 ge->URPU = carry_into_15;
132
133 *reg = result;
134
135 /* CC: overflow flag = FA04 = carry_out; nonzero flag = FA05 = (result!=0) */
136 uint8_t cc = (uint8_t)((carry_out << 1) | (result != 0 ? 1 : 0));
137 alu_set_cc(ge, cc);
138}

140/* =========================================================================
141 * SMR — Subtract Memory from Register (0xBF)
142 * §5.6.4.3: register = register - memory_value.
143 * Result is in 2's-complement form; negative stays complemented.
144 *
145 * CC encoding (§5.6.4.3) — SIGNED-result table:
146 * FA04 FA05 cc meaning
147 * 0 0 0 "Not possible" — never set
148 * 0 1 1 result < 0 (complemented) → ALU_CC_NEG (1)
149 * 1 0 2 result == 0 → ALU_CC_ZERO (2)
150 * 1 1 3 result > 0 → ALU_CC_POS (3)
151 *
152 * For signed interpretation we compare as signed 16-bit integers.
153 * The subtraction is implemented as addition of 2's complement of value,
154 * consistent with the GE-130 hardware.
155 *
156 * Carry flip-flops:
157 * URPE = borrow out of bit 15 (set when result wrapped, i.e. reg < value unsigned)
158 * URPU = borrow into bit 15
159 *
160 * Confidence: HIGH (direct table §5.6.4.3)
161 * =========================================================================*/
162void alu_smr(struct ge *ge, uint16_t *reg, uint16_t value)
163{
164 uint16_t result = (uint16_t)(*reg - value);
165
166 /*
167 * Borrow: in subtract-as-add-complement, carry out = NOT borrow.
168 * Unsigned borrow out: reg < value → borrow (URPE=1).
169 */
170 uint8_t borrow_out = (*reg < value) ? 1 : 0;
171
172 /*
173 * Borrow into bit 15: borrow from lower 15 bits.
174 * = 1 if (reg & 0x7FFF) < (value & 0x7FFF)
175 */
176 uint8_t borrow_into_15 = (uint8_t)((*reg & 0x7FFF) < (value & 0x7FFF) ? 1 : 0);
177
178 ge->URPE = borrow_out;
179 ge->URPU = borrow_into_15;
180
181 *reg = result;
182
183 /*
184 * CC based on signed 16-bit interpretation of result:
185 * NOTE: §5.6.4.3 says "result < 0 (complemented)" for FA05=1.
186 * Signed comparison:
187 */
188 uint8_t cc;
189 int16_t sresult = (int16_t)result;
190
191 if (sresult < 0)
192 cc = ALU_CC_NEG; /* 1 */
193 else if (sresult == 0)
194 cc = ALU_CC_ZERO; /* 2 */
195 else
196 cc = ALU_CC_POS; /* 3 */
197
198 alu_set_cc(ge, cc);
199}

213/* =========================================================================
214 * SR — Search Right (0xD9)
215 * §5.5.4.1: scan field left-to-right for model byte; deposit result in
216 * register 7 (r7 pointer supplied by caller).
217 *
218 * Result address:
219 * Found at offset i: r7 = field + i + 1
220 * Not found: r7 = field + len
221 *
222 * Qualitative result: "not interested" per §5.5.4.
223 *
224 * UNCERTAINTY: The manual says "The address 11 is the one to the left"
225 * (i.e. field parameter = leftmost address). Confirmed: field is leftmost.
226 * Confidence: MEDIUM (OCR of §5.5.4.1 truncated; logic is symmetric to SL)
227 * =========================================================================*/
228void alu_sr(struct ge *ge, uint16_t *r7,
229 uint16_t field, uint16_t len, uint8_t model)
230{
231 uint16_t i;
232
233 for (i = 0; i < len; i++) {
234 if (ge->mem[(uint16_t)(field + i)] == model) {
235 *r7 = (uint16_t)(field + i + 1);
236 return;
237 }
238 }
239
240 /* Not found: address following the field */
241 *r7 = (uint16_t)(field + len);
242}

244/* =========================================================================
245 * SL — Search Left (0xDB)
246 * §5.5.4.2: scan field right-to-left for model byte; deposit result in r7.
247 *
248 * Result address (symmetric to SR, direction right-to-left):
249 * Found at offset i (0=leftmost): r7 = field + i - 1
250 * (address "preceding" the match in left direction; field-1 if i==0)
251 * Not found: r7 = field - 1
252 *
253 * Qualitative result: "not interested" per §5.5.4.
254 *
255 * UNCERTAINTY: The manual says the address stored is "the one of the character
256 * following the end of the research field in the direction of the research."
257 * For SL the direction is right-to-left, so "following end" = field - 1.
258 * For a found character the symmetric definition gives field + i - 1.
259 * This needs hardware verification; behaviour at i==0 (field-1) may wrap.
260 * Confidence: MEDIUM
261 * =========================================================================*/
262void alu_sl(struct ge *ge, uint16_t *r7,
263 uint16_t field, uint16_t len, uint8_t model)
264{
265 uint16_t i;
266
267 /* Scan right-to-left: start at rightmost byte (field + len - 1) */
268 for (i = len; i > 0; i--) {
269 uint16_t pos = (uint16_t)(field + i - 1); /* current byte position */
270 if (ge->mem[pos] == model) {
271 /* "address preceding the match" in the left direction */
272 *r7 = (uint16_t)(field + i - 2); /* = pos - 1 */
273 return;
274 }
275 }
276
277 /* Not found: one before the field start */
278 *r7 = (uint16_t)(field - 1);
279}

281/* =========================================================================
282 * MVQ — Move Quartets (0xF8)
283 * §5.5.1.3, §3.084/3.098: Copy digit nibbles (low 4 bits) from src to dst.
284 * Zones (high 4 bits) of destination bytes are preserved.
285 * Both fields addressed from rightmost byte; operation proceeds right-to-left.
286 *
287 * CC result (§5.10.2):
288 * FA04=0 FA05=0 (cc=0) → all digit nibbles of result are zero
289 * FA04=0 FA05=1 (cc=1) → at least one digit nibble nonzero
290 *
291 * NOTE: MVQ uses the ADD-magnitude raw values 0 and 1 directly — these do NOT
292 * map to the COMPARE/SIGNED-result named constants; use raw literals here.
293 *
294 * UNCERTAINTY: "Zones are not processed" (§3.098) could mean:
295 * (a) zones of src are ignored (only digit nibbles copied) — implemented here
296 * (b) zones of dst are zeroed, entire byte overwritten
297 * (c) entire byte is copied (zones "not processed" meaning "not arithmetically
298 * significant")
299 * Interpretation (a) is most consistent with "decimal unpacked" semantics.
300 * Hardware trace recommended.
301 * Confidence: MEDIUM (OCR partially clear; semantics from §3.098 context)
302 * =========================================================================*/
303void alu_mvq(struct ge *ge, uint16_t dst, uint16_t src, uint8_t len)
304{
305 uint8_t any_nonzero = 0;
306 uint8_t i;
307
308 /* Process right-to-left; addresses given are already rightmost */
309 for (i = 0; i < len; i++) {
310 uint16_t sa = (uint16_t)(src - i);
311 uint16_t da = (uint16_t)(dst - i);
312 uint8_t digit = ge->mem[sa] & 0x0F; /* take digit nibble from src */
313 uint8_t zone = ge->mem[da] & 0xF0; /* keep zone nibble from dst */
314 ge_mem_store8(ge, da, zone | digit);
315 if (digit != 0)
316 any_nonzero = 1;
317 }
318
319 /*
320 * CC: 0 = all-zero result, 1 = nonzero result (ADD-magnitude raw values).
321 * FA04=0 always for MVQ (only FA05 encodes the result).
322 * Use raw 0/1 — these are not the COMPARE/SIGNED-result named constants.
323 */
324 alu_set_cc(ge, any_nonzero ? 1u : 0u);
325}

327/* =========================================================================
328 * CMQ — Compare Quartets (0xF9)
329 * §5.5.1.4: Compare digit nibbles of two fields. Neither altered.
330 * "Not considering the zones."
331 * Both fields addressed from rightmost byte.
332 * Comparison is left-to-right (most significant digit first),
333 * consistent with treating the field as a multi-digit number.
334 *
335 * CC result (§5.5.1.4) — COMPARE table:
336 * FA04 FA05 cc meaning
337 * 0 0 0 "Not possible" — never set
338 * 0 1 1 OP1 < OP2 → ALU_CC_LOW (1)
339 * 1 0 2 OP1 == OP2 → ALU_CC_EQUAL (2)
340 * 1 1 3 OP1 > OP2 → ALU_CC_HIGH (3)
341 *
342 * UNCERTAINTY: Same as MVQ regarding zone handling.
343 * Confidence: MEDIUM-HIGH (§5.5.1.4 text and §5.10.2 table consistent)
344 * =========================================================================*/
345void alu_cmq(struct ge *ge, uint16_t a, uint16_t b, uint8_t len)
346{
347 uint8_t i;
348
349 /*
350 * Compare left-to-right (most significant digit first).
351 * Addresses a and b are the rightmost bytes, so the leftmost
352 * byte of field 'a' is at (a - len + 1).
353 */
354 for (i = 0; i < len; i++) {
355 uint8_t da = ge->mem[(uint16_t)(a - (len - 1) + i)] & 0x0F;
356 uint8_t db = ge->mem[(uint16_t)(b - (len - 1) + i)] & 0x0F;
357
358 if (da < db) {
359 alu_set_cc(ge, ALU_CC_LOW); /* 1 */
360 return;
361 }
362 if (da > db) {
363 alu_set_cc(ge, ALU_CC_HIGH); /* 3 */
364 return;
365 }
366 }
367
368 /* All digit nibbles equal */
369 alu_set_cc(ge, ALU_CC_EQUAL); /* 2 */
370}
