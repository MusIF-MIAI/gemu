1/*
2 * printer.c - Integrated printer / console-typewriter peripheral (channel 2).
3 *
4 * See printer.h for the model rationale. Summary of the completion mechanism,
5 * derived empirically against the funktionalcpu report_and_end PER (0x19D0,
6 * `PER 0x00,0x19EE`) and the channel-2 / TPER flow charts:
7 *
8 * A channel-2 print/typewriter PER runs the CPU through the external-operation
9 * organisation phase (c8 d8 d9 .. a8 a9 aa ab) and parks it at state b8, the
10 * org-phase external request-wait. There the CPU suspends: RC00 drops, so the
11 * NA knot yields 0 (idle) and rSA stays 0 while rSO holds b8 pending. gemu has
12 * no channel-2 timing model to satisfy the wait, so it hangs.
13 *
14 * The faithful exit is gated on DU97 = PUC2 ^ L2.3 (signals.h). With PUC2
15 * asserted (and RC00 raised so NA_knot routes rSO=b8 into rSA), state_b8's own
16 * CU01/CU13/CU14/CU06 commands build the PER-completion future_state and the
17 * sequencer returns to alpha at the post-PER instruction with PO/registers
18 * intact (verified: it then runs the real final-verify CMC at 0x19d4 cleanly).
19 *
20 * Discriminating the channel-2 print-wait from the channel-1 reader input-wait:
21 * State b8 is SHARED with the card reader (its inter-byte "waiting for next
22 * column" gap looks identical: rSO=b8, rSA=0, RASI=1, lu08=0, RC00=0). The
23 * reader peripheral presents a byte within a couple of cycles, breaking the
24 * wait; a genuine channel-2 print-wait persists. We therefore debounce with a
25 * stall counter: only after the wait has held unbroken for STALL_THRESHOLD
26 * on_clock calls do we treat it as a printer op and complete it. The reader's
27 * inter-byte gap never approaches that, so a concurrently-registered reader is
28 * never disturbed.
29 */

39/* Consecutive idle b8-wait cycles before we conclude it is a channel-2 print
40 * op and not a reader inter-byte gap. The reader presents within a few cycles;
41 * 256 is comfortably above any inter-byte gap yet imperceptible to the user. */
42#define STALL_THRESHOLD 256

44/* Bound on bytes read from the print order block per completed PER. A printer
45 * line is at most ~132 columns; cap well above that and stop at a NUL. */

53/* Is the machine parked in the channel-2 org-phase external request-wait?
54 * rSO=b8 pending, rSA=0 (CPU suspended/idle), CPU-active request not raised,
55 * and the integrated reader is not presenting a byte (so this is not a reader
56 * inter-byte gap that the reader is about to break). */
57static int in_channel2_print_wait(struct ge *ge)
58{
59 return ge->rSO == 0xb8 &&
60 ge->rSA == 0x00 &&
61 !ge->RC00 &&
62 ge->integrated_reader.lu08 == 0;
63}

65/* Best-effort capture of the print order block at completion: read up to
66 * CAPTURE_MAX bytes from the channel-2 operand addresser (rV2), rendering each
67 * through the GE 100-series internal graphic set (gecode.c) — the machine's own
68 * character code, NOT ASCII — and stopping at a NUL. Non-graphic codes render
69 * '.'. Honest about its limits — see printer.h. */
70static void capture_line(struct ge *ge)
71{
72 struct ge_integrated_printer *p = &ge->integrated_printer;
73 uint16_t addr = ge->rV2;
74 int n = 0;
75
76 for (n = 0; n < CAPTURE_MAX; n++) {
77 uint16_t a = (uint16_t)(addr + n);
78 uint8_t b = ge->mem[a];
79 if (b == 0x00)
80 break;
81 if (p->out_len >= (int)sizeof(p->out) - 2)
82 break;
83 p->out[p->out_len++] = ge_glyph(b);
84 }
85 if (p->out_len < (int)sizeof(p->out) - 1)
86 p->out[p->out_len++] = '\n';
87 p->out[p->out_len] = '\0';
88
89 ge_log(LOG_PERI,
90 "printer: completed channel-2 PER, captured %d byte(s) from V2=0x%04x\n",
91 n, addr);
92}

94/*
95 * on_clock: called at TO00, the first clock of every machine cycle, before the
96 * MSL chart for the cycle runs. We watch for the channel-2 print-wait; once it
97 * has stalled past the debounce threshold we capture the line and assert the
98 * unit-ready / CPU-active conditions so the machine's own state_b8 microcode
99 * completes the PER (B8 -> alpha) on the next cycle.
100 */
101static int printer_on_clock(struct ge *ge, void *opaque)
102{
103 struct printer_ctx *ctx = (struct printer_ctx *)opaque;
104 struct ge_integrated_printer *p = &ge->integrated_printer;
105
106 if (!p->present)
107 return 0;
108
109 /* Channel-2 OUTPUT transfer engine: drive one character per cycle. While
110 * bytes remain, hold the channel-2 request (RC02) and the rSI output state
111 * so NA_knot routes this cycle to state_02 (mem[V4] -> RO -> CE16 -> sink).
112 * RC02 is asserted at TO00, before pulse() latches RIA2 = RC02. When the
113 * count is exhausted, drop the request and end the line. */
114 if (p->out_active) {
115 if (p->out_remaining > 0) {
116 ge->RC02 = 1;
117 ge->rSI = 0x02;
118 p->out_remaining--;
119 } else {
120 /* Transfer done: drop the request (so this cycle is no longer a
121 * channel-2 cycle and the CPU resumes), restore the CPU sequencer
122 * the stolen cycles clobbered, and end the line. */
123 ge->RC02 = 0;
124 ge->rSO = p->out_saved_so;
125 if (p->out_len < (int)sizeof(p->out) - 1) {
126 p->out[p->out_len++] = '\n'; /* end-of-print (models 0A/0B/FIRU) */
127 p->out[p->out_len] = '\0';
128 }
129 p->out_active = 0;
130 }
131 return 0;
132 }
133
134 if (!in_channel2_print_wait(ge)) {
135 ctx->stall = 0;
136 return 0;
137 }
138
139 if (++ctx->stall < STALL_THRESHOLD)
140 return 0;
141
142 /* Genuine channel-2 print-wait: capture + complete via the native path. */
143 capture_line(ge);
144 ge->PUC2 = 1; /* channel-2 unit ready -> DU97 = 1 (PUC2 ^ L2.3) */
145 ge->RC00 = 1; /* CPU-active request -> NA_knot routes rSO=b8 into rSA */
146 ctx->stall = 0; /* one-shot; rSO leaves b8 next cycle */
147 return 0;
148}

157/* Channel-2 OUTPUT sink: the CPU's channel-2 transfer microcode hands one
158 * character to the printer here (command CE16 "Load Printer Buffer", rSI state
159 * 02/03). The byte is the machine's internal GE graphic code; render it through
160 * the GE 100-series glyph table into the paper-feed buffer (same buffer the
161 * front-ends drain). See docs/peripherals.md "CAN2 data-transfer phase". */
162static void printer_sink(struct ge *ge, struct ge_channel *ch, uint8_t c)
163{
164 struct ge_integrated_printer *p = &ge->integrated_printer;
165 (void)ch;
166
167 if (p->out_len < (int)sizeof(p->out) - 1)
168 p->out[p->out_len++] = ge_glyph(c);
169 p->out[p->out_len] = '\0';
170}

172int printer_register(struct ge *ge)
173{
174 struct printer_ctx *ctx = calloc(1, sizeof(*ctx));
175 if (!ctx)
176 return -1;
177
178 ge->channel2.sink = printer_sink;
179 ge->integrated_printer.present = 1;
180 ge->integrated_printer.out_len = 0;
181 ge->integrated_printer.out[0] = '\0';
182 ge->integrated_printer.kbd_head = 0;
183 ge->integrated_printer.kbd_tail = 0;
184
185 ctx->stall = 0;
186 ctx->peri.next = NULL;
187 ctx->peri.init = NULL;
188 ctx->peri.on_pulse = NULL;
189 ctx->peri.on_clock = printer_on_clock;
190 ctx->peri.deinit = printer_deinit;
191 ctx->peri.ctx = ctx;
192
193 return ge_register_peri(ge, &ctx->peri);
194}

205{
206 struct ge_integrated_printer *p = &ge->integrated_printer;

208 if (next == p->kbd_head)
209 return; /* queue full; drop */
210 p->kbd[p->kbd_tail] = c;
211 p->kbd_tail = next;
